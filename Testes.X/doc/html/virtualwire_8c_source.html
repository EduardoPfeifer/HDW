146 {
147  uint8_t i;
148  uint8_t index = 0;
149  uint16_t crc = 0xffff;
150  uint8_t *p = vw_tx_buf;
151  uint8_t count = len + 3; // Added byte count and FCS to get total number of bytes
152 
153  if (len > VW_MAX_PAYLOAD)
154  return 0;
155 
156  // Wait for transmitter to become available
157  vw_wait_tx();
158  vw_rx_enabled = 0; // disable the receiver so it does corrupt our message
159 
160  // Encode the message length
161  crc = _crc_ccitt_update(crc, count);
162  p[index++] = symbols[count >> 4];
163  p[index++] = symbols[count & 0xf];
164 
165  // Encode the message into 6 bit symbols. Each byte is converted into
166  // 2 6-bit symbols, high nybble first, low nybble second
167  for (i = 0; i < len; i++)
168  {
169  crc = _crc_ccitt_update(crc, buf[i]);
170  p[index++] = symbols[buf[i] >> 4];
171  p[index++] = symbols[buf[i] & 0xf];
172  }
173 
174  // Append the fcs, 16 bits before encoding (4 6-bit symbols after encoding)
175  // Caution: VW expects the _ones_complement_ of the CCITT CRC-16 as the FCS
176  // VW sends FCS as low byte then hi byte
177  crc = ~crc;
178  p[index++] = symbols[(crc >> 4) & 0xf];
179  p[index++] = symbols[crc & 0xf];
180  p[index++] = symbols[(crc >> 12) & 0xf];
181  p[index++] = symbols[(crc >> 8) & 0xf];
182 
183  // Total number of 6-bit symbols to send
184  vw_tx_len = index + VW_HEADER_LEN;
185

187  vw_tx_start();
188 
189  return 1;

227 void vw_pll(void)
228 {
229  // Integrate each sample
230  if (vw_rx_sample)
231  vw_rx_integrator++;
232 
233  if (vw_rx_sample != vw_rx_last_sample)
234  {
235  // Transition, advance if ramp > 80, retard if < 80
236  vw_rx_pll_ramp += ((vw_rx_pll_ramp < VW_RAMP_TRANSITION)
237  ? VW_RAMP_INC_RETARD
238  : VW_RAMP_INC_ADVANCE);
239  vw_rx_last_sample = vw_rx_sample;
240  }
241  else
242  {
243  // No transition
244  // Advance ramp by standard 20 (== 160/8 samples)
245  vw_rx_pll_ramp += VW_RAMP_INC;
246  }
247  if (vw_rx_pll_ramp >= VW_RX_RAMP_LEN)
248  {
249  // Add this to the 12th bit of vw_rx_bits, LSB first
250  // The last 12 bits are kept
251  vw_rx_bits >>= 1;
252 
253  // Check the integrator to see how many samples in this cycle were high.
254  // If < 5 out of 8, then its declared a 0 bit, else a 1;
255  if (vw_rx_integrator >= 5)
256  vw_rx_bits |= 0x800;
257 
258  vw_rx_pll_ramp -= VW_RX_RAMP_LEN;
259  vw_rx_integrator = 0; // Clear the integral for the next cycle
260 
261  if (vw_rx_active)
262  {
263  // We have the start symbol and now we are collecting message bits,
264  // 6 per symbol, each which has to be decoded to 4 bits
265  if (++vw_rx_bit_count >= 12)
266  {
267  // Have 12 bits of encoded message == 1 byte encoded
268  // Decode as 2 lots of 6 bits into 2 lots of 4 bits
269  // The 6 lsbits are the high nybble
270  uint8_t this_byte =
271  (vw_symbol_6to4(vw_rx_bits & 0x3f)) << 4
272  | vw_symbol_6to4(vw_rx_bits >> 6);
273 
274  // The first decoded byte is the byte count of the following message
275  // the count includes the byte count and the 2 trailing FCS bytes
276  // REVISIT: may also include the ACK flag at 0x40
277  if (vw_rx_len == 0)
278  {
279  // The first byte is the byte count
280  // Check it for sensibility. It cant be less than 4, since it
281  // includes the bytes count itself and the 2 byte FCS
282  vw_rx_count = this_byte;
283  if (vw_rx_count < 4 || vw_rx_count > VW_MAX_MESSAGE_LEN)
284  {
285  // Stupid message length, drop the whole thing
286  vw_rx_active = 0;
287  return;
288  }
289  }
290  vw_tx_buf[vw_rx_len++] = this_byte;
291 
292  if (vw_rx_len >= vw_rx_count)
293  {
294  // Got all the bytes now
295  vw_rx_active = 0;
296  vw_rx_done = 1; // Better come get it before the next one starts
297  }
298  vw_rx_bit_count = 0;
299  }
300  }
301  // Not in a message, see if we have a start symbol
302  else if (vw_rx_bits == 0xb38)
303  {
304  // Have start symbol, start collecting message
305  vw_rx_active = 1;
306  vw_rx_bit_count = 0;
307  vw_rx_len = 0;
308  vw_rx_done = 0; // Too bad if you missed the last message
309  }
310  }
311 }
