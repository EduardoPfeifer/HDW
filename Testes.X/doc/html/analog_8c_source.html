58 uint16_t analog_read_interrupt_get_value() {
59  // Verifica se tá no modo de Sinal + Magnitude. Neste modo, 10 á 12 bits são a magnitude, 1 bit é o sinal e o resto é 0. Ex: mmmm mmmm mmmm 000s
60  if( ADCON1bits.ADFM == analog_result_format_sign_magnitude ) {
61  // Verifica se estava usando 12 ou 10 bits, e determina o shift que irá fazer com base nisso. Vide tabela 17-2.
62  uint8_t shift = ADCON0bits.ADRMD == analog_result_mode_bit_12 ? 16-12 : 16-10; // 4 ou 6, respectivamente.
63  // Recupera o resultado da conversão, junto com dados adicionais, como o sinal. Ex: -2355 = 1001 0011 0011 0001 -> bit 0 é o sinal.
64  uint16_t result = ADRES;
65  // Separa o sinal do resultado todo. Ex: -2355 = 1001 0011 0011 0001 -> 1
66  uint8_t sign = result & 0x0001;
67  // Desloca o resultado para direita, para ficar no valor que foi lido mesmo. Ex: 1001 0011 0011 0001 >> 4 -> 0000 1001 0011 0011
68  result = result >> shift;
69  // Se tinha o sinal de negativo, deixa o resultado final como negativo.
70  if( sign ) {
71  result = -result;
72  }
73  // finalmente, retorna o resultado.
74  return result;
75  // Tá no modo de Complemento de 2. Se o bit mais significativo for 1, tem que inverter todos os bits e somar +1 para tirar a resposta correta.
76  } else {
77  // Recupera o resultado da conversão. O formato é: xxxx mmmm mmmm mmmm ou
78  // xxxx xxmm mmmm mmmm, onde se x for 1, é porque o valor é negativo e está em complemento de 2.
79  uint16_t result = ADRES;
80  if( result & 0x8000 ) {
81  // Complemento de 1
82  result = ~result;
83  // Complemento de 2
84  result++;
85  // Deixa o valor negativo.
86  result = -result;
87  }
88  return result;
